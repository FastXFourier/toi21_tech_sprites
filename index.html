มี Graph อันนึงมี 1≤n≤1061 \leq n \leq 10^61≤n≤106 nodes (บ้านของ Tech-Sprites ที่ iii)
มี 1≤m≤3⋅1061 \leq m \leq 3 \cdot 10^61≤m≤3⋅106 edges
แต่ละ node iii มีค่า 2 ค่าคือ: a[i],b[i]a[i], b[i]a[i],b[i]

ใน connected-component (connected-component คือเซตของ node ที่ node ทุกตัวในเซตสามารถไปหากันได้โดยใช้ edges ที่ให้มา และไม่มี node อื่นภายนอกเซตที่สามารถเชื่อมถึงเซตนี้ได้)
เมื่อมีภารกิจสำคัญ ตัว Tech-Sprites ที่อยู่ใน connected-component จะออกมาตามลำดับ โดยตัวที่มีค่า a[i]a[i]a[i] น้อยกว่าจะออกมาก่อน
แต่ถ้า a[i]a[i]a[i] เท่ากัน b[i]b[i]b[i] น้อยกว่าจะออกมาก่อน

จงหาว่าต้องเพิ่ม edge อย่างน้อยกี่ edge เพื่อให้ Tech Sprites สามารถออกมาตามลำดับที่ถูกต้องได้
(จะออกมาทีละ connected-component และเรียงจาก a[i]a[i]a[i] ก่อน แล้ว b[i]b[i]b[i])

วิธีทำ

  1. เก็บข้อมูล a[i],b[i],ia[i], b[i], ia[i],b[i],i ไว้ใน array แล้ว sort ตามลำดับการออก (คือเรียงตาม a[i],b[i]a[i], b[i]a[i],b[i])
  2. ใช้ DSU (Disjoint Set Union) เพื่อหา connected-components
  3. map node เดิมไปยังลำดับที่ถูก sort แล้ว
  4. เราจะ loop จาก 111 ถึง N−1N - 1N−1 โดยเราจะเก็บ array ไว้ 2 อัน ชื่อว่า
    • sz[]sz[]sz[] โดย sz[i]sz[i]sz[i] จะเก็บว่า connected-component ที่มี iii เป็น root มี node ใน connected-component กี่ตัว
    • cnt[]cnt[]cnt[] โดย cnt[i]cnt[i]cnt[i] จะเก็บว่า ตั้งแต่ loop มา มี tech-sprites ที่มี root อยู่ที่ node iii รวมทั้งหมดกี่ตัว (รวมตัวที่เราเชื่อมมาใหม่ด้วย)
  5. ถ้าหากว่าจำนวนตัวที่อยู่ใน root ของตัวปัจจุบันยังมีจำนวนตัวไม่ครบ (sz[dsu(i)](sz[dsu(i)](sz[dsu(i)] != cnt[dsu(i)])cnt[dsu(i)])cnt[dsu(i)]) เราจะต้องตรวจว่าตัวถัดไปในลำดับมี root เป็นตัวเดียวกันหรือไม่ ถ้าไม่ ต้องเชื่อมด้วย edge เพิ่ม (unite node iii และ i+1i + 1i+1 ด้วย) และเพิ่มค่า ansansans ไป 1

และภาพแสดงตำแหน่งที่เราตรวจสอบเงื่อนไขในข้อ 5 แล้วไม่ตรงเงื่อนไข จึงจะต้องเพิ่มเส้นเชื่อมเข้าไป:

Code

#include <bits/stdc++.h>
using namespace std;

#define int long long
const int N = 1e6 + 5, M = 3e6 + 5;

int n, m, par[N], sz[N], cnt[N], mp[N];
tuple<int, int, int> a[N];

int dsu(int a) {
    return par[a] = (a == par[a] ? a : dsu(par[a]));
}

void unite(int u, int v) {
    int uu = dsu(u), vv = dsu(v);
    if (uu != vv) {
        sz[vv] += sz[uu];
        cnt[vv] += cnt[uu];
        par[uu] = vv;
    }
}

int32_t main() {
    cin.tie(NULL)->sync_with_stdio(false);
    for (auto &e : sz) e = 1;
    iota(par, par + N, 0ll);

    cin >> n >> m;

    for (int i = 1; i <= n; i++) {
        auto &[x, y, idx] = a[i];
        cin >> x >> y;
        idx = i;
    }

    sort(a + 1, a + 1 + n);

    for (int i = 1; i <= n; i++) {
        auto &[x, y, idx] = a[i];
        mp[idx] = i;
    }

    for (int i = 1; i <= m; i++) {
        int u, v;
        cin >> u >> v;
        unite(mp[u], mp[v]);
    }

    int ans = 0;
    for (int i = 1; i < n; i++) {
        cnt[dsu(i)]++;
        if (cnt[dsu(i)] == sz[dsu(i)]) continue;
        if (dsu(i + 1) != dsu(i)) {
            ans++;
            unite(i, i + 1);
        }
    }

    cout << ans;
}
