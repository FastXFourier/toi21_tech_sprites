คำอธิบายวิธีทำพร้อม code สำหรับข้อ Tech Sprites

Link: https://otog.in.th/problem/1086

โจทย์แบบคร่าวๆ

มี Graph อันนึงมี 1≤n≤1061 \leq n \leq 10^61≤n≤106 nodes(บ้านของ Tech-Sprites ที่ iii) มี 1≤m≤3⋅1061\leq m\leq 3\cdot10^61≤m≤3⋅106 edges แต่ละ node ที่ iii มีค่า 2 ค่าคือ: a[i],b[i]a[i],b[i]a[i],b[i] โดยใน connected-component(connected-component เป็นเซตของ node ที่ node ทุก node ในเซตสามารถไปหากันโดยใช้ edges ที่ให้มาและไม่มี node vvv ที่ vvv สามารถไปหา node หนึ่งในเซตได้) โดยเมื่อมีภารกิจสำคัญตัว Tech-Sprites ที่อยู่ใน connected-component จะออกมาตามลำดับโดยตัวที่มีค่า a[i]a[i]a[i] น้อยกว่าจะออกมาก่อน แต่ถ้า a[i]a[i]a[i] เท่ากันตัว b[i]b[i]b[i] จะได้ออกมาก่อน โดยต้องพิมพ์ว่าต้องอย่างน้อยเพิ่ม edge กี่ edge เพื่อให้ Tech Sprites สามารถออกมาตามลำดับที่ถูกต้องได้(จะออกมาทีละ connected-component และ a[i]a[i]a[i] น้อยกว่าจะต้องมาก่อน ถ้า a[i]a[i]a[i] เท่ากัน b[i]b[i]b[i] น้อยกว่าจะต้องมาก่อน)


เช่นในเคสนี้ไม่จำเป็นต้องเพิ่ม edge เพราะสามารถทำให้ Tech Sprites ออกมาตามลำดับโดยไม่ต้องเพิ่ม edge ได้

แต่ในเคสนี้เราจำเป็นต้องเพิ่ม edge 1 edge เพราะไม่สามารถทำให้ Tech Sprites ออกมาตามลำดับโดยไม่ต้องเพิ่ม edge ได้ แต่ถ้าเพิ่ม 1 edge จะทำให้ Tech Sprites ออกมาตามลำดับได้

(ใส่ edge แบบนี้ได้)

solution:

ต้องมี array vvv ของ Tech Sprites ที่มีการ sort ตาม a[i],b[i]a[i],b[i]a[i],b[i] ก่อนว่าลำดับที่ออกมาต้องออกมาอย่างไร(โดยเก็บด้วยว่าอยู่ที่ node ไหน) แล้วใช้ DSU(Disjoint Set Union) ในการหา connected-component ต่างๆ จากนั้นก็ไปตามแต่ละตัวของ array vvv โดยตัวที่เป็น DSU เดียวกันจะต้องอยู่กันทั้งหมด(เพื่อให้ออกมาตามลำดับได้ถูกต้อง) ถ้าพบว่ามีบางช่วงไม่ติดกันก็ต้องเพิ่ม edge ไปเชื่อมกับ DSU ที่ทำให้บางช่วงไม่ติดกัน


(โดยจากรูปตัว array vvv จะเก็บเป็น {a[u],b[u],u}\{a[u],b[u],u\}{a[u],b[u],u} คือมีค่าว่าอยู่ node uuu และมีค่า a[u],b[u]a[u],b[u]a[u],b[u])

โดยจุดที่มีลูกศรสีแดงทำให้ DSU ไม่ติดกันทำให้ต้องเพิ่ม edge อย่างน้อย 2 edges

Solution code:

#include <bits/stdc++.h>
using namespace std;
#define int long long
const int N = 1e6 + 5,M = 3e6 + 5;
int n, m, par[N], sz[N], cnt[N], mp[N];
tuple <int, int, int> a[N];
int dsu(int a){
	return par[a] = (a == par[a] ? a : dsu(par[a]));
}
void unite(int u, int v){ int uu = dsu(u), vv = dsu(v);
	if (uu != vv) {
		sz[vv] += sz[uu];
		cnt[vv] += cnt[uu];
		par[uu] = vv;
	}
}
int32_t main(){
	cin.tie(NULL)->sync_with_stdio(false);
	for (auto &e : sz) e = 1;
	iota(par, par + N, 0ll);
	cin >> n >> m;
	for (int i = 1; i <= n; i++) {
		auto &[x, y, idx] = a[i];
		cin >> x >> y; idx = i;
	}
	sort(a + 1, a + 1 + n);
	for (int i = 1; i <= n; i++) {
		auto &[x, y, idx] = a[i];
		mp[idx] = i;
	}
	for (int i = 1; i <= m; i++) {
		int u, v; cin >> u >> v;
		unite(mp[u], mp[v]);
	}
	int ans = 0;
	for (int i = 1; i < n; i++) {
		cnt[dsu(i)]++;
		if (cnt[dsu(i)] == sz[dsu(i)]) continue;
		if (dsu(i + 1) != dsu(i)) {
			ans++;
			unite(i, i + 1);
		}
	}
	cout << ans;
}

Total Time Complexity: O(n)O(n)O(n)
